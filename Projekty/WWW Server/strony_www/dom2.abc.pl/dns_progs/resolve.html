#include <arpa/inet.h>
#include <netdb.h>
#include <stdio.h>

int main(int argc, char **argv) {
	struct hostent *hp = gethostbyname(argv[1]);
	if (hp == NULL) {
		printf("gethostbyname() failed\n");
	} else {
		for (int i=0; hp->h_addr_list[i] != NULL; i++) 
			printf("%s\n", inet_ntoa( *( struct in_addr*)( hp -> h_addr_list[i])));
	}
}
